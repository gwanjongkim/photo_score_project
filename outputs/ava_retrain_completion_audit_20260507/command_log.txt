import csv, glob
for p in sorted(glob.glob("data/processed/ava/*unit*.csv")):
    with open(p, 'r') as f:
        reader = csv.reader(f)
        header = next(reader)
        if "aesthetic_unit_score" in header:
            idx = header.index("aesthetic_unit_score")
            vals = []
            for row in reader:
                try: vals.append(float(row[idx]))
                except: pass
            if vals: print(p, "min", min(vals), "max", max(vals))
